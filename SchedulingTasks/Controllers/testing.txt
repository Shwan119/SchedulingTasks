-- Store IDs first
DECLARE @IDsToDelete TABLE (ID INT);

INSERT INTO @IDsToDelete (ID)
SELECT t1.ID 
FROM MainTable t1
INNER JOIN Table2 t2 ON t1.SomeID = t2.ID
INNER JOIN Table3 t3 ON t2.AnotherID = t3.ID
WHERE t3.Status = 'Inactive';

-- Verify count
SELECT COUNT(*) FROM @IDsToDelete;

-- Delete
DELETE FROM MainTable 
WHERE ID IN (SELECT ID FROM @IDsToDelete);
